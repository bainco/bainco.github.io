 36def _get_coordinates_by_dimension(canvas, tag, dimension='x'):
 37    '''
 38    updates the x and y position of all shapes that have been tagged
 39    with the tag argument
 40    '''
 41    if dimension == 'x':
 42        pos = 0
 43    else:
 44        pos = 1
 45    shape_ids = canvas.find_withtag(tag)
 46    coords = []
 47    for id in shape_ids:
 48        shape_coords = _get_coordinates(canvas, id)
 49        for i in range(0, len(shape_coords)):
 50            if i % 2 == pos:
 51                coords.append(shape_coords[i])
 52    return coords

107def make_poly_oval(canvas, center, radius_x, radius_y, fill_color='#FF4136', tag=None, stroke_width=1, outline=None):
108    x, y = center
109    x0, y0, x1, y1 = (x - radius_x, y - radius_y, x + radius_x, y + radius_y)
110
111    steps = 100
112    # major and minor axes
113    a = (x1 - x0) / 2.0
114    b = (y1 - y0) / 2.0
115
116    # center
117    xc = x0 + a
118    yc = y0 + b
119
120    point_list = []
121
122    # create the oval as a list of points
123    for i in range(steps):
124        # Calculate the angle for this step
125        theta = (math.pi * 2) * (float(i) / steps)
126
127        x = a * math.cos(theta)
128        y = b * math.sin(theta)
129
130        point_list.append(round(x + xc))
131        point_list.append(round(y + yc))
132
133    return canvas.create_polygon(
134        point_list,
135        fill=fill_color,
136        width=stroke_width,
137        tags=tag,
138        outline=outline,
139        smooth=True
140    )

142def rotate(canvas, tag, degrees=5, origin=None):
143    '''
144    Rotates any polygons with a given tag around a point on a given canvas. Note that this does
145    not work with circles or ovals generated using the `create_oval`/`create_circle` functions.
146
147    * `canvas` (`Canvas`): [Required] The `Canvas` to draw on.
148    * `tag` (`str`): The tag of the object to animate.
149    * `degrees` (`int`): How far, in degrees, you would like to rotate the object.
150    * `origin` (`tuple`): A coordinate about which to rotate.
151    '''
152    if origin is None:
153        # calculate reasonable origin
154        top = get_top(canvas, tag)
155        bottom = get_bottom(canvas, tag)
156        left = get_left(canvas, tag)
157        right = get_right(canvas, tag)
158        origin = (right - left, bottom - top)
159
160    degrees = math.radians(degrees)
161    ox, oy = origin
162
163    shape_ids = canvas.find_withtag(tag)
164    for id in shape_ids:
165        coords = _get_coordinates(canvas, id)
166        # update coordinates:
167        for i in range(0, len(coords), 2):
168            px, py = coords[i], coords[i+1]
169            qx = ox + math.cos(degrees) * (px - ox) - math.sin(degrees) * (py - oy)
170            qy = oy + math.sin(degrees) * (px - ox) + math.cos(degrees) * (py - oy)
171            coords[i] = qx
172            coords[i+1] = qy
173        # set the coordinates:
174        _set_coordinates(canvas, id, coords)

334def make_image(
335        canvas, image_path, position=(200, 200), rotation=None,
336        scale=None, **kwargs):
337    '''
338    Draws a given image on the screen. NOTE: Requires the `pillow` package to be installed.

340    * `canvas` (`Canvas`): [Required] The `Canvas` to drawn on.

348    anchor='nw'

351    dir_path = os.path.dirname(os.path.realpath(__file__))

353    pil_image = Image.open(image_path)
354    print(scale)
355    if scale:
356        size = (
357            round(pil_image.size[0] * scale),
358            round(pil_image.size[1] * scale)
359        )
360        pil_image = pil_image.resize(size)
361    if rotation:
362        pil_image = pil_image.rotate(rotation)  # note: returns a copy

428def flip(canvas, tag):
429    '''
430    Flip a given tagged object on a particular canvas.
431
432    * `canvas` (`Canvas`): [Required] The `Canvas` to drawn on.
433    * `tag` (`str`): [Required] The tag of the object to flip.
434    '''
435    center = get_center(canvas, tag)
436    width = get_width(canvas, tag)
437    shape_ids = canvas.find_withtag(tag)
438    for shape_id in shape_ids:
439        flipped_coordinates = []
440        shape_coords = _get_coordinates(canvas, shape_id)
441        counter = 0
442        for num in shape_coords:
443            if counter % 2 == 0:
444                if num < center:
445                    flipped_coordinates.append(num + 2 * (center - num))
446                elif num > center:
447                    flipped_coordinates.append(num - 2 * (num - center))
448                else:
449                    flipped_coordinates.append(num)
450            else:
451                flipped_coordinates.append(num)
452            counter += 1
453        _set_coordinates(canvas, shape_id, flipped_coordinates)

456def make_grid(canvas, width, height):
457    '''

459
460    * `canvas` (`Canvas`): [Required] The `Canvas` to drawn on.
461    * `width` (`int`): [Required] Width of the grid to draw.

463    '''
464    interval = 100
465
466    # Delete old grid if it exists:
467    canvas.delete('grid_line')
468    # Creates all vertical lines at intevals of 100
469    for i in range(0, width, interval):
470        canvas.create_line(i, 0, i, height, tag='grid_line')
471
472        # Creates all horizontal lines at intevals of 100
473        for i in range(0, height, interval):
474            canvas.create_line(0, i, width, i, tag='grid_line')
475
476            # Creates axis labels
477            offset = 2
478            for y in range(0, height, interval):
479                for x in range(0, width, interval):
480                    canvas.create_oval(
481                    x - offset,
482                    y - offset,
483                    x + offset,
484                    y + offset,
485                    fill='black'
486                    )
487                    canvas.create_text(
488                    x + offset,
489                    y + offset,
490                    text="({0}, {1})".format(x, y),
491                    anchor="nw",
492                    font=("Purisa", 8)
493                    )

143def rotate(canvas, tag, degrees=5, origin=None):
144    '''
145    Rotates any polygons with a given tag around a point on a given canvas. Note that this does
146    not work with circles or ovals generated using the `create_oval`/`create_circle` functions.
147
148    * `canvas` (`Canvas`): [Required] The `Canvas` to draw on.
149    * `tag` (`str`): The tag of the object to animate.

151    * `origin` (`tuple`): A coordinate about which to rotate.
152    '''
153    if origin is None:
154        # calculate reasonable origin
155        top = get_top(canvas, tag)
156        bottom = get_bottom(canvas, tag)
157        left = get_left(canvas, tag)
158        right = get_right(canvas, tag)
159        origin = (right - left, bottom - top)
160
161    degrees = math.radians(degrees)
162    ox, oy = origin
163
164    shape_ids = canvas.find_withtag(tag)
165    for id in shape_ids:
166        coords = _get_coordinates(canvas, id)
167        # update coordinates:
168        for i in range(0, len(coords), 2):
169            px, py = coords[i], coords[i+1]
170            qx = ox + math.cos(degrees) * (px - ox) - math.sin(degrees) * (py - oy)
171            qy = oy + math.sin(degrees) * (px - ox) + math.cos(degrees) * (py - oy)
172            coords[i] = qx
173            coords[i+1] = qy
174        # set the coordinates:
175        _set_coordinates(canvas, id, coords)
